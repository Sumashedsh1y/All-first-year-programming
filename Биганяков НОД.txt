 int SearchNOD(int a,int b)
 {
     int q=0;
     if (a>b)
        for (int z=1;z<=b;z++)
        {
            if ((a % z == 0) && (b % z == 0))
                q=z;
        }
    if (a<b)
        for (int z=1;z<=a;z++)
        {
            if ((a % z == 0) && (b % z == 0))
                q=z;
        }
    return(q);
 }
 int EvclidIter(int a, int b)
 {
   int c;
   while (b) {
      c = a % b;
      a = b;
      b = c;
   }
   return (a);
 }
 int EvclidRecur(int a, int b)
{
    if (a==b)
        return a;
    else
    if (a>b)
        return EvclidRecur(a-b, b);
    else
        return EvclidRecur(a, b-a);
}
