#include <math.h>
#include <stdio.h>
#include <time.h>
#include <stdlib.h>
void selectionSort(int *mas, int size)
{
  int min, z;
  for (int i = 0; i < size - 1; i++)
  {
    min = i;
    for (int j = i + 1; j < size; j++)
    {
      if (mas[j] < mas[min])
        min = j;
    }
    z = mas[i];
    mas[i] = mas[min];
    mas[min] = z;
  }
}
void bubbleSort(int *mas, int size)
{
    int z;
    for (int i = 0; i < size - 1; i++)
    {
        for (int j = (size - 1); j > i; j--)
        {
        if (mas[j - 1] > mas[j])
        {
            z = mas[j - 1];
            mas[j - 1] = mas[j];
            mas[j] = z;
      }
    }
  }
}
void QSRT(int *mas, int size)
{
    int comp (const int *i, const int *j)
    {
        return *i - *j;
    }
    qsort(mas, 100, sizeof(int), comp);
}

     int mas[99];
