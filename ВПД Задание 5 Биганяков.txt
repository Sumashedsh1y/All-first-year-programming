#include <SFML/Graphics.hpp>

int main()
{
    sf::RenderWindow window(sf::VideoMode(500, 500), "SFML");
    sf::CircleShape shape(50.f);
    shape.setPosition(50, 50);
    sf::ConvexShape triangle;
    triangle.setPointCount(3);
    triangle.setPoint(0, sf::Vector2f(300, 250));
    triangle.setPoint(1, sf::Vector2f(350, 200));
    triangle.setPoint(2, sf::Vector2f(400, 250));
    sf::ConvexShape tail;
    tail.setPointCount(4);
    tail.setPoint(0, sf::Vector2f(275,450));
    tail.setPoint(1, sf::Vector2f(300,400));
    tail.setPoint(2, sf::Vector2f(400,400));
    tail.setPoint(3, sf::Vector2f(425,450));
    sf::RectangleShape rectangle(sf::Vector2f(100, 150));
    rectangle.setPosition(300, 250);
    sf::CircleShape wind(25.f);
    wind.setFillColor(sf::Color(0, 255, 255));
    wind.setPosition(325, 250);

    while (window.isOpen())
    {
        sf::Event event;
        while (window.pollEvent(event))
        {
            if (event.type == sf::Event::Closed)
                window.close();
        }

        window.clear(sf::Color(150,150,150));
        window.draw(shape);
        window.draw(triangle);
        window.draw(rectangle);
        window.draw(tail);
        window.draw(wind);
        window.display();
    }

    return 0;
}
