#include <math.h>
#include <stdio.h>
#include <locale.h>
float Recur(int);
float Inter(int);
int main()
{
    int  k;
    float x,y,z,w;
    printf("Vvedite k: ");
    scanf("%i",&k);
    y = Inter(k);
    z = Recur(k);
    printf("Interacional:%f", y);
    printf("\nRecurrent:%f", z);
    return 0;
}
float Inter(int k)
{
    float x, w = 0;
    for (x = 1; x <= k; x++)
    {
        w = w+(2*x-1);
    }
    return w;
}
float Recur(int k)
{
    if (k==0)
    {
        return k;
    }
    else
    {
        return Recur(k-1)+(2*k-1);
    }
}
